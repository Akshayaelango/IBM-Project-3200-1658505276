import random
from time import *
value=True
while(value):
    t=random.randint(0,50)
    h=random.randint(10,50)
    if t>45 and h<40:
        print("Temperature=",t,"Humidity=",h)
        print("ALARM ON")
        value=False
    else:
        print("Temperature=",t,"Humidity=",h)
    sleep(1)
